package my.packagez;

import org.javaparser.ast.Node;
import org.javaparser.ast.body.BodyDeclaration;
import org.javaparser.ast.body.ClassOrInterfaceDeclaration;
import org.javaparser.ast.body.MethodDeclaration;
import org.javaparser.ast.body.TypeDeclaration;
import org.javaparser.ast.expr.NameExpr;

import java.util.Optional;

/**
 * This class can be used to conveniently retrieve certain node from a JavaParser AST.
 */
public final class Navigator {

    private Navigator() {
        // prevent instantiation
    }

    public static Optional<TypeDeclaration> findType(CompilationUnit cu, String name) {
        return cu.getTypes().stream().filter((t) -> t.getName().equals(name)).findFirst();
    }

}
